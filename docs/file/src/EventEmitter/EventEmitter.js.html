    /**
 * @access private
 * this is EventEmitter description.
 * @example
 * let myClass = new EventEmitter();
 */
class EventEmitter {
    constructor() {
        this._eeEvents = {};
    }

    addListener(eventName, listener) {
        const self = this;

        if(!self._eeEvents[eventName]) {
            self._eeEvents[eventName] = [];
        }

        self._eeEvents[eventName].push(listener);

        return self;
    }

    emit(eventName, args) {
        const self = this;

        const eventListeners = self._eeEvents[eventName];

        if(eventListeners) {
            eventListeners.forEach( listener => listener.call(self, args));
        }

        return this;
    }
}

export default EventEmitter;
